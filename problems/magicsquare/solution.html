    3. def ismagic(amagic):
    sumset=set()
    sum = 0

    #check column:
    for i in range(3):
        for j in range(3):
            sum += amagic[j][i]
        sumset.add(sum)
        sum=0
    #check rows
    for i in range(3):
        for j in range(3):
            sum += amagic[i][j]
        sumset.add(sum)
        sum = 0
    #check easy diag
    for i in range(3):
        sum += amagic[i][i]
    sumset.add(sum)
    sum = 0
    #check hard diag
    for i in range(3):
        sum += amagic[i][2 - i]
    sumset.add(sum)
    sum = 0
    return len(sumset) == 1
